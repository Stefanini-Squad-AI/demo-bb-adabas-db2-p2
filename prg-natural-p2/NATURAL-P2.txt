DEFINE DATA 
LOCAL
* --- DBATDP-1: Sócio data via COBOL/DB2 (STFSCC00C consult, STFSCC00I insert).
* --- LDA mirror: prg-natural-p2/LOCAL/STFSCCSOC.txt
1 SOCIO-COMMAREA
  2 SOCIO-CN-RC (A2)
  2 SOCIO-CN-RG-NUM (N9)
  2 SOCIO-CN-NOME (A40)
  2 SOCIO-CN-DATA-CADASTRO (A10)
  2 SOCIO-CN-CATG (N2)
  2 SOCIO-CN-INDI-DIVIDA (A1)
  2 SOCIO-CN-DATA-BAIXA (A10)
  2 SOCIO-CN-HORA-BAIXA (A12)
  2 SOCIO-CN-OBSV (A500)
  2 SOCIO-CN-SUPER1 (A20)
  2 SOCIO-CN-PAG-ROW (1:12)
    3 SOCIO-CN-PAG-DATA-VENC (A10)
    3 SOCIO-CN-PAG-VALR (P6.2)
    3 SOCIO-CN-PAG-OK (A1)
*
* Legacy MAP 'MAPSOCIO' field names (no direct ADABAS view).
1 #MAPA
  2 #RG-CONSULTA (N7) 
  2 #DIA-VENCIAMENTO (N2)
  2 NOME-SOCIO-PRINCIPAL (A40)
  2 CATG-SOCIO (N2)
  2 OBSV-CLIENTE (A500)
  2 VALR-MENSALIDADE (P6.2) (1:12)
  2 DATA-VENCIMENTO (A10) (1:12)
  2 PAGAMENTO-OK (L) (1:12)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (N4)
  2 #MES (N2)
  2 #DIA (N2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* Legacy ADABAS path (retired):
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
*   IF NO
*     PERFORM INCLUI-SOCIO
*     ESCAPE BOTTOM 
*   END-NOREC
*   #TEXTO := 'Sócio já cadastrado.'
* END-FIND
*
RESET SOCIO-CN-RC
SOCIO-CN-RG-NUM := #RG-CONSULTA
CALL 'STFSCC00C' SOCIO-COMMAREA
DECIDE ON FIRST VALUE SOCIO-CN-RC
  VALUE 'EX'
    #TEXTO := 'Sócio já cadastrado.'
  VALUE 'NF'
    PERFORM INCLUI-SOCIO
  NONE VALUE
    COMPRESS 'Erro na consulta ao DB2. RC=' SOCIO-CN-RC INTO #TEXTO
END-DECIDE

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
SOCIO-CN-RG-NUM := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE EDITED *DATX TO SOCIO-CN-DATA-CADASTRO (EM=YYYY-MM-DD)
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*) 
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE TO SOCIO-CN-PAG-DATA-VENC(#INDICE) (EM=9999-99-99)
  MOVE EDITED #DATA-HOJE TO DATA-VENCIMENTO(#INDICE) (EM=9999-99-99)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
MOVE ' ' TO SOCIO-CN-INDI-DIVIDA
MOVE ALL ' ' TO SOCIO-CN-DATA-BAIXA
MOVE ALL ' ' TO SOCIO-CN-HORA-BAIXA
MOVE ALL ' ' TO SOCIO-CN-SUPER1
*
IF OBSV-CLIENTE EQ ' '
  MOVE 'Novo sócio' TO SOCIO-CN-OBSV
ELSE
  MOVE OBSV-CLIENTE TO SOCIO-CN-OBSV
END-IF
MOVE NOME-SOCIO-PRINCIPAL TO SOCIO-CN-NOME
MOVE CATG-SOCIO TO SOCIO-CN-CATG
FOR #INDICE = 1 TO 12
  MOVE VALR-MENSALIDADE(#INDICE) TO SOCIO-CN-PAG-VALR(#INDICE)
  IF PAGAMENTO-OK(#INDICE)
    MOVE 'Y' TO SOCIO-CN-PAG-OK(#INDICE)
  ELSE
    MOVE 'N' TO SOCIO-CN-PAG-OK(#INDICE)
  END-IF
END-FOR
*
* Legacy ADABAS path (retired):
* STORE
RESET SOCIO-CN-RC
CALL 'STFSCC00I' SOCIO-COMMAREA
DECIDE ON FIRST VALUE SOCIO-CN-RC
  VALUE 'OK'
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 'DU'
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'Erro na inclusão ao DB2. RC=' SOCIO-CN-RC INTO #TEXTO
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
