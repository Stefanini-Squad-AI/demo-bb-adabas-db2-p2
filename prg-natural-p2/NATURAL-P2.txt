DEFINE DATA 
LOCAL
INCLUDE STFSCSOC
*
*>>> ADABAS legado: substituído por chamadas COBOL STFSC00C/STFSC00I + DB2.
*    Manter book LOCAL em prg-natural/LOCAL/STFSCSOC na FUSER/biblioteca.
*
* 1 SOCIO VIEW OF ADABAS-SOCIOS
*   2 NUMB-SOCIO-PRINCIPAL
*   2 NOME-SOCIO-PRINCIPAL
*   2 DATA-CADASTRO
*   2 C*PERIODICO-PAGAMENTO
*   2 VALR-MENSALIDADE (12)
*   2 DATA-VENCIMENTO (12)
*   2 PAGAMENTO-OK (12)
*   2 CATG-SOCIO
*   2 INDI-DIVIDA
*   2 DATA-BAIXA
*   2 HORA-BAIXA
*   2 OBSV-CLIENTE
*   2 SUPER1
1 SOCIO
  2 NUMB-SOCIO-PRINCIPAL (N9)
  2 NOME-SOCIO-PRINCIPAL (A40)
  2 DATA-CADASTRO (D)
  2 VALR-MENSALIDADE (P6.2/1:12)
  2 DATA-VENCIMENTO (D/1:12)
  2 PAGAMENTO-OK (L/1:12)
  2 CATG-SOCIO (N2)
  2 INDI-DIVIDA (A1)
  2 DATA-BAIXA (D)
  2 HORA-BAIXA (T)
  2 OBSV-CLIENTE (A500)
  2 SUPER1 (A50)
*
1 #MAPA
  2 #RG-CONSULTA (N7) 
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
RESET #STFSC-LNK-C
#LNK-C-RG-NUMB := #RG-CONSULTA
CALL 'STFSC00C' #STFSC-LNK-C
*
*>>> ADABAS legado:
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
*   IF NO
*     PERFORM INCLUI-SOCIO
*     ESCAPE BOTTOM 
*   END-NOREC
*   #TEXTO := 'Sócio já cadastrado.'
* END-FIND
*
DECIDE FOR FIRST VALUE OF #LNK-C-RC
  VALUE 100
    PERFORM INCLUI-SOCIO
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'Erro consulta STFSC00C rc' #LNK-C-RC INTO #TEXTO
END-DECIDE

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
DATA-CADASTRO := *DATX
*
* Mensalidades: mantém CALLNAT externo (não versionado neste repositório).
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*) 
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  DATA-VENCIMENTO(#INDICE) := #DATA-HOJE
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET INDI-DIVIDA DATA-BAIXA HORA-BAIXA
INDI-DIVIDA := 'N'
*
IF OBSV-CLIENTE EQ ' '
  OBSV-CLIENTE := 'Novo sócio'
END-IF
*
RESET #STFSC-LNK-I
#LNK-I-RG-NUMB := NUMB-SOCIO-PRINCIPAL
#LNK-I-NOME := NOME-SOCIO-PRINCIPAL
MOVE EDITED DATA-CADASTRO TO #LNK-I-DATA-CADASTRO (EM=YYYY-MM-DD)
#LNK-I-CATG-SOCIO := CATG-SOCIO
#LNK-I-INDI-DIVIDA := INDI-DIVIDA
#LNK-I-DATA-BAIXA := ' '
#LNK-I-HORA-BAIXA := ' '
#LNK-I-OBSV-SOCIO := OBSV-CLIENTE
#LNK-I-SUPER1 := SUPER1
#LNK-I-QTD-PAG := 12
FOR #INDICE = 1 TO 12
  MOVE EDITED DATA-VENCIMENTO(#INDICE) TO #LNK-I-PAG-DATA-VENC(#INDICE) (EM=YYYY-MM-DD)
  #LNK-I-PAG-VALR(#INDICE) := VALR-MENSALIDADE(#INDICE)
  IF PAGAMENTO-OK(#INDICE)
    #LNK-I-PAG-OK(#INDICE) := 'Y'
  ELSE
    #LNK-I-PAG-OK(#INDICE) := 'N'
  END-IF
END-FOR
*
CALL 'STFSC00I' #STFSC-LNK-I
DECIDE FOR FIRST VALUE OF #LNK-I-RC
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 110
    #TEXTO := 'RG já cadastrado (duplicidade em DB2).'
  NONE VALUE
    COMPRESS 'Erro inclusão STFSC00I rc' #LNK-I-RC INTO #TEXTO
END-DECIDE
*
*>>> ADABAS legado:
* STORE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
