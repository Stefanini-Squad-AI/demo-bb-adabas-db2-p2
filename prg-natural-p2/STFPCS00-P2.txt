DEFINE DATA
LOCAL USING STFSC00L
*
* MIGRACAO DBATDP-18: VIEW ADABAS-SOCIOS substituida por LDA STFSC00L/COBOL-DB2
* 1 SOCIO VIEW OF ADABAS-SOCIOS
*   2  NUMB-SOCIO-PRINCIPAL
*   2  NOME-SOCIO-PRINCIPAL
*   2  DATA-CADASTRO
*   2  C*PERIODICO-PAGAMENTO
*   2  VALR-MENSALIDADE (12)
*   2  DATA-VENCIMENTO (12)
*   2  PAGAMENTO-OK (12)
*   2  CATG-SOCIO
*   2  INDI-DIVIDA
*   2  DATA-BAIXA
*   2  HORA-BAIXA
*   2  OBSV-CLIENTE
*   2  SUPER1
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* MIGRACAO DBATDP-18: FIND ADABAS substituido por CALLNAT STFSC00C (COBOL/DB2)
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
RESET STFSC00-RETORNO
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALLNAT 'STFSC00C' STFSC00-AREA
DECIDE ON FIRST VALUE OF STFSC00-SQLCODE
  VALUE 100
    PERFORM INCLUI-SOCIO
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2' STFSC00-SQLCODE STFSC00-MSG INTO #TEXTO
END-DECIDE

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
* MIGRACAO DBATDP-18: conversao data Natural D para padrao DB2 YYYY-MM-DD
MOVE EDITED *DATX TO DATA-CADASTRO (EM=YYYY-MM-DD)
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  COMPRESS #ANO '-' #MES '-' #DIA INTO DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET INDI-DIVIDA DATA-BAIXA HORA-BAIXA
*
IF OBSV-CLIENTE EQ ' '
  OBSV-CLIENTE := 'Novo sócio'
END-IF
*
C-PERIODICO-PAGAMENTO := 12
*
* MIGRACAO DBATDP-18: STORE ADABAS substituido por CALLNAT STFSC00I (COBOL/DB2)
* STORE
RESET STFSC00-RETORNO
CALLNAT 'STFSC00I' STFSC00-AREA
DECIDE ON FIRST VALUE OF STFSC00-SQLCODE
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 803
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    IF STFSC00-MSG NE ' '
      #TEXTO := STFSC00-MSG
    ELSE
      COMPRESS 'ERRO DB2' STFSC00-SQLCODE INTO #TEXTO
    END-IF
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
