DEFINE DATA
LOCAL
USING 'STFPCS00'
1 #MAPA
  2 #RG-CONSULTA (N9)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
*
1 #DATA-CADASTRO-FORMATADA (A10)
1 #DATA-VENCIMENTO-FORMATADA (A10)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* === ALTERAÇÃO: Substituição de FIND direto em ADABAS ===
* === por chamada a COBOL/DB2 (STFSC00C) ===
MOVE 'C' TO STFPCS00-OPERACAO
MOVE #RG-CONSULTA TO SOCIOS-RG

CALLNAT 'STFSC00C' STFPCS00-AREA

* === Avaliação do retorno do COBOL ===
IF STFPCS00-RETURN-CODE = 100
  PERFORM INCLUI-SOCIO
  ESCAPE BOTTOM
ELSE IF STFPCS00-RETURN-CODE = 0
  #TEXTO := 'Sócio já cadastrado.'
ELSE
  #TEXTO := 'Erro ao consultar sócio em DB2.'
END-IF
END-IF

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
MOVE #RG-CONSULTA TO SOCIOS-RG

IF SOCIOS-NOME EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF SOCIOS-CATG
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK SOCIOS-CATG
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK SOCIOS-CATG
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
* === ALTERAÇÃO: Formatação de data para DB2 (YYYY-MM-DD) ===
MOVE *DATX (EM=YYYYMMDD) TO #DATA-HOJE
#DATA-CADASTRO-FORMATADA :=
  #ANO || '-' || #MES || '-' || #DIA
MOVE #DATA-CADASTRO-FORMATADA TO SOCIOS-DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' SOCIOS-CATG PAGO-VALOR(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
MOVE 1 TO PAGO-STATUS(1)
RESET PAGO-STATUS(2:11)
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE #ANO || '-' || #MES || '-' || #DIA
    TO #DATA-VENCIMENTO-FORMATADA
  MOVE #DATA-VENCIMENTO-FORMATADA
    TO PAGO-DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    MOVE '01' TO #MES
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET SOCIOS-DIVIDA SOCIOS-DATA-BAIXA SOCIOS-HORA-BAIXA
*
IF SOCIOS-OBSV EQ ' '
  MOVE 'Novo sócio' TO SOCIOS-OBSV
END-IF
*
* === ALTERAÇÃO: Substituição de STORE direto em ADABAS ===
* === por chamada a COBOL/DB2 (STFSC00I) ===
MOVE 'I' TO STFPCS00-OPERACAO

CALLNAT 'STFSC00I' STFPCS00-AREA

* === Avaliação do retorno do COBOL ===
IF STFPCS00-RETURN-CODE = 0
  #TEXTO := 'Novo sócio incluído com sucesso.'
ELSE IF STFPCS00-RETURN-CODE = 803
  #TEXTO := 'RG já existe. Verifique os dados.'
ELSE
  #TEXTO := 'Erro ao incluir sócio em DB2.'
END-IF
END-IF
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
