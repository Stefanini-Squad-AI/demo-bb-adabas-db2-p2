DEFINE DATA
LOCAL USING STFSC00LD
*
* Campos de tela / negocio (substituem VIEW ADABAS-SOCIOS)
1 SOCIO-TELA
  2 NUMB-SOCIO-PRINCIPAL   (N9)
  2 NOME-SOCIO-PRINCIPAL   (A40)
  2 DATA-CADASTRO          (A10)
  2 C-PERIODICO-PAGAMENTO  (N3)
  2 VALR-MENSALIDADE       (P6.2/12)
  2 DATA-VENCIMENTO        (A10/12)
  2 PAGAMENTO-OK           (A1/12)
  2 CATG-SOCIO             (N2)
  2 INDI-DIVIDA            (A1)
  2 DATA-BAIXA             (A10)
  2 HORA-BAIXA             (A5)
  2 OBSV-CLIENTE           (A500)
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINE #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #DATA-ISO (A10)
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* --- ADABAS (substituido): FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
RESET STFSC00-AREA-COMUNICACAO
STFSC00-ACAO := 'C'
STFSC00-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALLNAT 'STFSC00C' STFSC00-AREA-COMUNICACAO
*
DECIDE ON FIRST VALUE OF STFSC00-SQLCODE
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  VALUE 100
    PERFORM INCLUI-SOCIO
  VALUE 803
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2' STFSC00-SQLCODE INTO #TEXTO
    REINPUT #TEXTO
END-DECIDE
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE EDITED *DATX (EM=YYYY-MM-DD) TO DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := 'Y'
FOR #INDICE = 2 TO 12
  PAGAMENTO-OK(#INDICE) := 'N'
END-FOR
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE (EM=YYYY-MM-DD) TO DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
INDI-DIVIDA := 'N'
RESET DATA-BAIXA HORA-BAIXA
*
IF OBSV-CLIENTE EQ ' '
  OBSV-CLIENTE := 'Novo sócio'
END-IF
*
C-PERIODICO-PAGAMENTO := 12
*
* --- ADABAS (substituido): STORE
RESET STFSC00-AREA-COMUNICACAO
STFSC00-ACAO := 'I'
PERFORM MOVE-TELA-PARA-COBOL
CALLNAT 'STFSC00I' STFSC00-AREA-COMUNICACAO
*
DECIDE ON FIRST VALUE OF STFSC00-SQLCODE
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 803
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2' STFSC00-SQLCODE INTO #TEXTO
    REINPUT #TEXTO
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
DEFINE MOVE-TELA-PARA-COBOL
STFSC00-NUMB-SOCIO-PRINCIPAL := NUMB-SOCIO-PRINCIPAL
STFSC00-NOME-SOCIO-PRINCIPAL := NOME-SOCIO-PRINCIPAL
STFSC00-DATA-CADASTRO        := DATA-CADASTRO
STFSC00-CATG-SOCIO           := CATG-SOCIO
STFSC00-INDI-DIVIDA          := INDI-DIVIDA
STFSC00-DATA-BAIXA           := DATA-BAIXA
STFSC00-HORA-BAIXA           := HORA-BAIXA
STFSC00-OBSV-SOCIO           := OBSV-CLIENTE
STFSC00-QTD-PERIODICO        := C-PERIODICO-PAGAMENTO
FOR #INDICE = 1 TO C-PERIODICO-PAGAMENTO
  STFSC00-SEQ-OCORRENCIA(#INDICE)   := #INDICE
  STFSC00-DATA-VENCIMENTO(#INDICE)  := DATA-VENCIMENTO(#INDICE)
  STFSC00-VALR-MENSALIDADE(#INDICE) := VALR-MENSALIDADE(#INDICE)
  STFSC00-PAGAMENTO-OK(#INDICE)     := PAGAMENTO-OK(#INDICE)
END-FOR
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
