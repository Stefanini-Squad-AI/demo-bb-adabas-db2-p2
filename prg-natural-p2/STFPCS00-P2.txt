DEFINE DATA
LOCAL USING STFSC00
*
* --- ADABAS (migração DBATDP-18): acesso direto desativado ---
* 1 SOCIO VIEW OF ADABAS-SOCIOS
*   2  NUMB-SOCIO-PRINCIPAL
*   2  NOME-SOCIO-PRINCIPAL
*   2  DATA-CADASTRO
*   2  C*PERIODICO-PAGAMENTO
*   2  VALR-MENSALIDADE (12)
*   2  DATA-VENCIMENTO (12)
*   2  PAGAMENTO-OK (12)
*   2  CATG-SOCIO
*   2  INDI-DIVIDA
*   2  DATA-BAIXA
*   2  HORA-BAIXA
*   2  OBSV-CLIENTE
*   2  SUPER1
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* --- Consulta via COBOL/DB2 (substitui FIND ADABAS-SOCIOS) ---
RESET STFSC00-AREA
STFSC00-OPERACAO := 'C'
STFSC00-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALL 'STFSC00C' STFSC00-AREA
*
DECIDE FOR FIRST CONDITION
  WHEN STFSC00-RETURN-CODE EQ 0
    #TEXTO := 'Sócio já cadastrado.'
  WHEN STFSC00-RETURN-CODE EQ 100
    PERFORM INCLUI-SOCIO
  WHEN NONE
    COMPRESS 'ERRO ' STFSC00-RETURN-CODE ' NA CONSULTA DB2' INTO #TEXTO
END-DECIDE
*
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
*   IF NO
*     PERFORM INCLUI-SOCIO
*     ESCAPE BOTTOM
*   END-NOREC
*   #TEXTO := 'Sócio já cadastrado.'
* END-FIND
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
STFSC00-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF STFSC00-NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF STFSC00-CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *STFSC00-CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *STFSC00-CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE EDITED *DATX (EM=YYYYMMDD) TO STFSC00-DATA-CADASTRO
  WITH MASK 'YYYY-MM-DD'
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' STFSC00-CATG-SOCIO STFSC00-VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
STFSC00-PAGAMENTO-OK(1) := 'Y'
STFSC00-PAGAMENTO-OK(2:12) := 'N'
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE TO STFSC00-DATA-VENCIMENTO(#INDICE)
    WITH MASK 'YYYY-MM-DD'
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
STFSC00-C-PERIODICO-PAGAMENTO := 12
STFSC00-INDI-DIVIDA := 'N'
RESET STFSC00-DATA-BAIXA STFSC00-HORA-BAIXA
*
IF STFSC00-OBSV-SOCIO EQ ' '
  STFSC00-OBSV-SOCIO := 'Novo sócio'
END-IF
*
* --- Inclusão via COBOL/DB2 (substitui STORE ADABAS-SOCIOS) ---
STFSC00-OPERACAO := 'I'
CALL 'STFSC00I' STFSC00-AREA
*
DECIDE FOR FIRST CONDITION
  WHEN STFSC00-RETURN-CODE EQ 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  WHEN STFSC00-RETURN-CODE EQ 803
    #TEXTO := 'Sócio já cadastrado.'
  WHEN NONE
    COMPRESS 'ERRO ' STFSC00-RETURN-CODE ' NA INCLUSÃO DB2' INTO #TEXTO
END-DECIDE
*
* STORE
* #TEXTO := 'Novo sócio incluído com sucesso.'
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
