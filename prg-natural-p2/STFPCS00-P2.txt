DEFINE DATA
LOCAL
LOCAL USING 'STFSC00'
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
*
1 #DATA-FORMATADA (A10)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* ORIGINAL: FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
* NOW REPLACED WITH CALLNAT STFSC00C for DB2 via COBOL
MOVE 'C' TO OPERACAO
MOVE #RG-CONSULTA TO NUMB-SOCIO-PRINCIPAL
CALLNAT 'STFSC00C' STFSC00-AREA

EVALUATE RETURN-CODE
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  VALUE 100
    PERFORM INCLUI-SOCIO
    ESCAPE BOTTOM
  VALUE OTHER
    COMPRESS 'Erro ao consultar sócio. Código: ' RETURN-CODE
        INTO #TEXTO
END-EVALUATE

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
DATA-CADASTRO := *DATX
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*) 
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE WITH MASK 'YYYY-MM-DD'
      TO #DATA-FORMATADA
  MOVE #DATA-FORMATADA TO DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET DATA-BAIXA HORA-BAIXA
*
IF OBSV-SOCIO EQ ' '
  MOVE 'Novo sócio' TO OBSV-SOCIO
END-IF
*
* ORIGINAL: STORE to ADABAS
* NOW REPLACED WITH CALLNAT STFSC00I for DB2 via COBOL
MOVE 'I' TO OPERACAO
MOVE DATA-CADASTRO TO #DATA-FORMATADA
CALLNAT 'STFSC00I' STFSC00-AREA

EVALUATE RETURN-CODE
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE OTHER
    COMPRESS 'Erro ao incluir sócio. Código: ' RETURN-CODE
        INTO #TEXTO
    REINPUT #TEXTO
END-EVALUATE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
