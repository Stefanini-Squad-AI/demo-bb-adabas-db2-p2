DEFINE DATA
LOCAL USING STFSC00L
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
*
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* --- Migracao ADABAS -> COBOL/DB2 (DBATDP-18) ---
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
*   IF NO
*     PERFORM INCLUI-SOCIO
*     ESCAPE BOTTOM
*   END-NOREC
*   #TEXTO := 'Sócio já cadastrado.'
* END-FIND
*
STFSC00L-OPERACAO := 'C'
STFSC00L-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALL 'STFSC00C' STFSC00L-AREA
*
DECIDE FOR FIRST VALUE STFSC00L-RETORNO
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  VALUE 100
    PERFORM INCLUI-SOCIO
  VALUE 803
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2 ' STFSC00L-RETORNO ' NA CONSULTA.' INTO #TEXTO
END-DECIDE
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
STFSC00L-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF STFSC00L-NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF STFSC00L-CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *STFSC00L-CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *STFSC00L-CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
#DATA-VENC-D := *DATX
MOVE EDITED #DATA-VENC-D (EM=YYYYMMDD) TO STFSC00L-DATA-CADASTRO
  WITH MASK='YYYY-MM-DD'
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' STFSC00L-CATG-SOCIO STFSC00L-VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
STFSC00L-PAGAMENTO-OK(1) := 'T'
STFSC00L-PAGAMENTO-OK(2:12) := 'F'
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE (EM=YYYYMMDD) TO STFSC00L-DATA-VENCIMENTO(#INDICE)
    WITH MASK='YYYY-MM-DD'
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
STFSC00L-INDI-DIVIDA := 'F'
RESET STFSC00L-DATA-BAIXA STFSC00L-HORA-BAIXA
*
IF STFSC00L-OBSV-SOCIO EQ ' '
  STFSC00L-OBSV-SOCIO := 'Novo sócio'
END-IF
*
STFSC00L-C-PERIODICO-PAGAMENTO := 12
*
* --- Migracao ADABAS -> COBOL/DB2 (DBATDP-18) ---
* STORE
*
STFSC00L-OPERACAO := 'I'
CALL 'STFSC00I' STFSC00L-AREA
*
DECIDE FOR FIRST VALUE STFSC00L-RETORNO
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 803
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2 ' STFSC00L-RETORNO ' NA INCLUSAO.' INTO #TEXTO
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
