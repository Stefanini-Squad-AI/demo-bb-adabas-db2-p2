DEFINE DATA 
LOCAL USING "STFSSOCIO"
*
1 #MAPA
  2 #RG-CONSULTA (N7) 
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* --- ADABAS FIND SOCIO (migrado para COBOL/DB2 via STFSC00C) ---
WS-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
RESET WS-RETORNO
CALLNAT 'STFSC00C' STFSSOCIO
*
DECIDE ON FIRST VALUE WS-RETORNO
  VALUE 100
    PERFORM INCLUI-SOCIO
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO na consulta DB2. Código:' WS-RETORNO INTO #TEXTO
END-DECIDE
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
WS-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF WS-NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *WS-CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *WS-CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *WS-CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE EDITED *DATX (EM=YYYY-MM-DD) TO WS-DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' WS-CATG-SOCIO WS-VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
WS-PAGAMENTO-OK(1) := 'T'
WS-PAGAMENTO-OK(2:12) := 'F'
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE (EM=YYYY-MM-DD)
    TO WS-DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
WS-INDI-DIVIDA := 'F'
WS-DATA-BAIXA := ' '
WS-HORA-BAIXA := ' '
*
IF WS-OBSV-SOCIO EQ ' '
  WS-OBSV-SOCIO := 'Novo sócio'
END-IF
*
WS-QTD-PAGAMENTO := 12
*
* --- ADABAS STORE SOCIO (migrado para COBOL/DB2 via STFSC00I) ---
RESET WS-RETORNO
CALLNAT 'STFSC00I' STFSSOCIO
*
DECIDE ON FIRST VALUE WS-RETORNO
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 803
    #TEXTO := 'Sócio já cadastrado (chave duplicada).'
  NONE VALUE
    COMPRESS 'ERRO na inclusão DB2. Código:' WS-RETORNO INTO #TEXTO
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
