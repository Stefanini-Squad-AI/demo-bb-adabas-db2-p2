DEFINE DATA
LOCAL USING "SOCIO-LOCAL"
1 SOCIO VIEW OF ADABAS-SOCIOS
  2  NUMB-SOCIO-PRINCIPAL
  2  NOME-SOCIO-PRINCIPAL
  2  DATA-CADASTRO
  2  C*PERIODICO-PAGAMENTO
  2  VALR-MENSALIDADE (12)
  2  DATA-VENCIMENTO (12)
  2  PAGAMENTO-OK (12)
  2  CATG-SOCIO
  2  INDI-DIVIDA
  2  DATA-BAIXA
  2  HORA-BAIXA
  2  OBSV-CLIENTE
  2  SUPER1
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* MIGRATED FROM ADABAS: FIND SOCIO with NUMB-SOCIO-PRINCIPAL
* Now using STFSC00C COBOL program with DB2 access
SO-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALLNAT 'STFSC00C' SOCIO-LKSP
*
IF SO-RETURN-CODE EQ 100
  * MIGRATED FROM ADABAS: PERFORM INCLUI-SOCIO
  * Record not found, proceed with inclusion
  PERFORM INCLUI-SOCIO
  ESCAPE BOTTOM
ELSE
  IF SO-RETURN-CODE EQ 0
    #TEXTO := 'Sócio já cadastrado.'
  ELSE
    #TEXTO := SO-MSG-ERROR
  END-IF
END-IF

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
SO-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF SO-NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF SO-CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *SO-CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *SO-CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
* MIGRATED FROM ADABAS: Convert to DB2 date format (YYYY-MM-DD)
MOVE EDITED *DATX (EM=YYYY-MM-DD) TO SO-DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' SO-CATG-SOCIO SO-VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
MOVE '1' TO SO-PAGAMENTO-OK(1)
PERFORM VARYING #INDICE FROM 2 BY 1 UNTIL #INDICE > 12
  MOVE '0' TO SO-PAGAMENTO-OK(#INDICE)
END-PERFORM
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  * MIGRATED FROM ADABAS: Convert to DB2 date format (YYYY-MM-DD)
  MOVE EDITED #DATA-HOJE (EM=YYYY-MM-DD) TO SO-DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET SO-DATA-BAIXA SO-HORA-BAIXA
*
IF SO-OBSV-SOCIO EQ ' '
  MOVE 'Novo sócio' TO SO-OBSV-SOCIO
END-IF
*
* MIGRATED FROM ADABAS: STORE to DB2 via STFSC00I COBOL program
CALLNAT 'STFSC00I' SOCIO-LKSP
*
IF SO-RETURN-CODE EQ 0
  #TEXTO := 'Novo sócio incluído com sucesso.'
ELSE
  IF SO-RETURN-CODE EQ 803
    #TEXTO := 'Sócio já existe na base de dados.'
  ELSE
    #TEXTO := 'Erro ao incluir sócio: ' SO-MSG-ERROR
  END-IF
END-IF
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
