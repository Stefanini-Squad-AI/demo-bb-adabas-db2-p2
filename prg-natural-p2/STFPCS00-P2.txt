DEFINE DATA
* MIGRATION: Substituído VIEW ADABAS por LOCAL DATA AREA SOCIO_LDA
LOCAL USING "SOCIO_LDA"
LOCAL
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
* MIGRATION: Variável para capturar return code de operações DB2
1 #RC-SOCIO (N4)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* MIGRATION: Substituído FIND ADABAS por chamada CALLNAT STFSC00C
MOVE #RG-CONSULTA TO NUMB-SOCIO-PRINCIPAL IN SOCIO-RECORD
MOVE 'C' TO OPERACAO IN SOCIO-RECORD
CALLNAT 'STFSC00C' SOCIO-RECORD

MOVE RETURN-CODE-DB2 IN SOCIO-RECORD TO #RC-SOCIO

IF #RC-SOCIO EQ 0
  #TEXTO := 'Sócio já cadastrado.'
ELSE
  IF #RC-SOCIO EQ 100
    PERFORM INCLUI-SOCIO
    IF #RC-SOCIO NOT EQ 0
      #TEXTO := 'Erro ao incluir sócio.'
    ELSE
      #TEXTO := 'Novo sócio incluído com sucesso.'
    END-IF
  ELSE
    #TEXTO := 'Erro ao consultar sócio.'
  END-IF
END-IF

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
* MIGRATION: Campo atribuído via SOCIO-RECORD (comunicação COBOL)
MOVE #RG-CONSULTA TO NUMB-SOCIO-PRINCIPAL IN SOCIO-RECORD

IF NOME-SOCIO-PRINCIPAL IN SOCIO-RECORD EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF CATG-SOCIO IN SOCIO-RECORD
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO IN SOCIO-RECORD
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO IN SOCIO-RECORD
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
* MIGRATION: Data de cadastro formatada em YYYY-MM-DD (padrão DB2)
MOVE EDITED *DATX (EM=YYYYMMDD) TO DATA-CADASTRO IN SOCIO-RECORD
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO IN SOCIO-RECORD
    VALR-MENSALIDADE(*) IN SOCIO-RECORD
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
MOVE 1 TO PAGAMENTO-OK(1) IN SOCIO-RECORD
MOVE 0 TO PAGAMENTO-OK(2:11) IN SOCIO-RECORD
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  * MIGRATION: Data de vencimento formatada em YYYY-MM-DD
  MOVE EDITED #DATA-HOJE (EM=YYYYMMDD) TO
      DATA-VENCIMENTO(#INDICE) IN SOCIO-RECORD
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
* MIGRATION: Campos de baixa inicializados (não aplicáveis em novo sócio)
MOVE SPACES TO DATA-BAIXA IN SOCIO-RECORD
MOVE SPACES TO HORA-BAIXA IN SOCIO-RECORD
*
IF OBSV-SOCIO IN SOCIO-RECORD EQ ' '
  MOVE 'Novo sócio' TO OBSV-SOCIO IN SOCIO-RECORD
END-IF
*
* MIGRATION: Substituído STORE ADABAS por chamada CALLNAT STFSC00I
MOVE 'I' TO OPERACAO IN SOCIO-RECORD
CALLNAT 'STFSC00I' SOCIO-RECORD

MOVE RETURN-CODE-DB2 IN SOCIO-RECORD TO #RC-SOCIO
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
