DEFINE DATA
LOCAL USING 'STFSOCIO'
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
*-- Migracao DBATDP-18: FIND Adabas substituido por STFSC00C (COBOL/DB2)
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
*   IF NO
*     PERFORM INCLUI-SOCIO
*     ESCAPE BOTTOM
*   END-NOREC
*   #TEXTO := 'Sócio já cadastrado.'
* END-FIND
*
WS-ACAO := 'C'
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALL 'STFSC00C' STFSOCIO-LINKAGE
*
DECIDE ON FIRST VALUE OF WS-RETORNO-CODIGO
  VALUE +100
    PERFORM INCLUI-SOCIO
    REINPUT #TEXTO
  VALUE +0
    #TEXTO := 'Sócio já cadastrado.'
    REINPUT #TEXTO
  NONE VALUE
    COMPRESS 'ERRO COBOL ' WS-RETORNO-CODIGO INTO #TEXTO
    REINPUT #TEXTO
END-DECIDE
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE EDITED *DATX (EM=YYYYMMDD) TO DATA-CADASTRO (EM=YYYY-MM-DD)
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := 'T'
PAGAMENTO-OK(2:12) := 'F'
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE (EM=YYYYMMDD)
    TO DATA-VENCIMENTO(#INDICE) (EM=YYYY-MM-DD)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
INDI-DIVIDA := 'F'
RESET DATA-BAIXA HORA-BAIXA
*
IF OBSV-SOCIO EQ ' '
  OBSV-SOCIO := 'Novo sócio'
END-IF
*
C-PERIODICO-PAGAMENTO := 12
*
*-- Migracao DBATDP-18: STORE Adabas substituido por STFSC00I (COBOL/DB2)
* STORE
*
WS-ACAO := 'I'
CALL 'STFSC00I' STFSOCIO-LINKAGE
*
IF WS-RETORNO-CODIGO = +0
  #TEXTO := 'Novo sócio incluído com sucesso.'
ELSE IF WS-RETORNO-CODIGO = +803
  #TEXTO := 'Sócio já cadastrado.'
  REINPUT #TEXTO
ELSE
  COMPRESS 'ERRO COBOL ' WS-RETORNO-CODIGO INTO #TEXTO
  REINPUT #TEXTO
END-IF
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
