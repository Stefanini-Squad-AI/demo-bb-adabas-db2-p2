DEFINE DATA
LOCAL USING "STFSC00-SOCIOS"
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #DATA-FORMATTED (A10)
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* ADABAS FIND operation — replaced by COBOL CALL STFSC00C
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALLNAT 'STFSC00C' SOCIOS-RECORD
*
EVALUATE RETURN-CODE-OPER
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  VALUE 100
    PERFORM INCLUI-SOCIO
    ESCAPE BOTTOM
  VALUE OTHER
    #TEXTO := 'Erro ao consultar sócio na base de dados.'
END-EVALUATE

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
*-- Convert registration date to YYYY-MM-DD format
DATA-CADASTRO := *DATX (EM=YYYYMMDD)
MOVE EDITED DATA-CADASTRO (EM=YYYY-MM-DD)
    TO DATA-CADASTRO.
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := 1
PAGAMENTO-OK(2:11) := 0
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #ANO #MES #DIA (EM=YYYY-MM-DD)
      TO DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET INDI-DIVIDA DATA-BAIXA HORA-BAIXA
*
IF OBSV-SOCIO EQ ' '
  OBSV-SOCIO := 'Novo sócio'
END-IF
*
* ADABAS STORE operation — replaced by COBOL CALL STFSC00I
CALLNAT 'STFSC00I' SOCIOS-RECORD
*
EVALUATE RETURN-CODE-OPER
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 300
    #TEXTO := 'Erro: sócio já existe na base de dados.'
  VALUE 500
    #TEXTO := 'Erro ao inserir sócio na base de dados.'
  VALUE OTHER
    #TEXTO := 'Erro desconhecido ao incluir sócio.'
END-EVALUATE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
