DEFINE DATA
LOCAL USING SOCIOS-LOCAL
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
RESET SOCIO-SQLCODE-DISP SOCIO-PERIODICO-CNT-IN SOCIO-PERIODICO-CNT-OUT
MOVE #RG-CONSULTA TO SOCIO-NUMB-PRINCIPAL
* [MIGRADO] Chamada para COBOL/DB2 (ex-FIND ADABAS SOCIO)
CALL 'STFSC00C' SOCIO-BOOK
IF SOCIO-RETURN-CODE = 1
  PERFORM INCLUI-SOCIO
ELSE
  #TEXTO := 'Sócio já cadastrado.'
END-IF

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
MOVE #RG-CONSULTA TO SOCIO-NUMB-PRINCIPAL
*
IF SOCIO-NOME-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF SOCIO-CATG
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *SOCIO-CATG
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *SOCIO-CATG
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
SOCIO-DATA-CADASTRO := #DATA-HOJE
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' SOCIO-CATG SOCIO-PER-VALR(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
SOCIO-PER-PAGO-OK(1) := 'Y'
SOCIO-PER-PAGO-OK(2:11) := 'N'
*
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  SOCIO-PER-DATA-VENC(#INDICE) := #DATA-HOJE
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET SOCIO-INDI-DIVIDA SOCIO-DATA-BAIXA SOCIO-HORA-BAIXA
*
IF SOCIO-OBSV EQ ' '
  SOCIO-OBSV := 'Novo sócio'
END-IF
*
MOVE 12 TO SOCIO-PERIODICO-CNT-IN
* [MIGRADO] Chamada para COBOL/DB2 (ex-STORE ADABAS SOCIO)
CALL 'STFSC00I' SOCIO-BOOK
IF SOCIO-RETURN-CODE NE 0
  COMPRESS 'Erro DB2 na inclusão. RC=' SOCIO-RETURN-CODE
    ' SQLCODE=' SOCIO-SQLCODE-DISP INTO #TEXTO
  REINPUT #TEXTO
END-IF
#TEXTO := 'Novo sócio incluído com sucesso.'
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
