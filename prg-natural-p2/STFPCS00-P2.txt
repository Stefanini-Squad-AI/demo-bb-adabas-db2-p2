DEFINE DATA
LOCAL USING STFPCS00LDA
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* >>> DBATDP-9: ponto de migração ADABAS (antes: FIND SOCIO ... ADABAS-SOCIOS)
MOVE 'C' TO STFSC00-OPERACAO
MOVE #RG-CONSULTA TO STFSC00-NUMB-SOCIO-PRINCIPAL
MOVE 0 TO STFSC00-RETURN-CODE
MOVE 0 TO STFSC00-SQLCODE-DSP
MOVE ' ' TO STFSC00-SQLSTATE
CALL 'STFSC00C' STFSC00-COMM-AREA
DECIDE ON FIRST VALUE STFSC00-RETURN-CODE
  VALUE 1
    PERFORM INCLUI-SOCIO
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2 SQLCODE' STFSC00-SQLCODE-DSP 'SQLSTATE' STFSC00-SQLSTATE INTO #TEXTO
END-DECIDE
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
DATA-CADASTRO := *DATX
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  DATA-VENCIMENTO(#INDICE) := #DATA-HOJE
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET INDI-DIVIDA DATA-BAIXA HORA-BAIXA
*
IF OBSV-CLIENTE EQ ' '
  OBSV-CLIENTE := 'Novo sócio'
END-IF
*
MOVE 'I' TO STFSC00-OPERACAO
MOVE NUMB-SOCIO-PRINCIPAL TO STFSC00-NUMB-SOCIO-PRINCIPAL
MOVE NOME-SOCIO-PRINCIPAL TO STFSC00-NOME-SOCIO-PRINCIPAL
MOVE EDITED DATA-CADASTRO TO STFSC00-DATA-CADASTRO (EM=YYYY-MM-DD)
MOVE CATG-SOCIO TO STFSC00-CATG-SOCIO
MOVE INDI-DIVIDA TO STFSC00-INDI-DIVIDA
MOVE '          ' TO STFSC00-DATA-BAIXA
MOVE '     ' TO STFSC00-HORA-BAIXA
MOVE OBSV-CLIENTE TO STFSC00-OBSV-CLIENTE
MOVE SUPER1 TO STFSC00-SUPER1
FOR #INDICE = 1 TO 12
  MOVE EDITED DATA-VENCIMENTO(#INDICE) TO STFSC00-DATA-VENCIMENTO(#INDICE) (EM=YYYY-MM-DD)
  MOVE VALR-MENSALIDADE(#INDICE) TO STFSC00-VALR-MENSALIDADE(#INDICE)
  IF PAGAMENTO-OK(#INDICE)
    MOVE 'Y' TO STFSC00-PAGAMENTO-OK(#INDICE)
  ELSE
    MOVE 'N' TO STFSC00-PAGAMENTO-OK(#INDICE)
  END-IF
END-FOR
MOVE 0 TO STFSC00-RETURN-CODE
MOVE 0 TO STFSC00-SQLCODE-DSP
MOVE ' ' TO STFSC00-SQLSTATE
CALL 'STFSC00I' STFSC00-COMM-AREA
IF STFSC00-RETURN-CODE NE 0
  COMPRESS 'ERRO DB2 SQLCODE' STFSC00-SQLCODE-DSP 'SQLSTATE' STFSC00-SQLSTATE INTO #TEXTO
  REINPUT #TEXTO
END-IF
#TEXTO := 'Novo sócio incluído com sucesso.'
*
* >>> DBATDP-9: ponto de migração ADABAS (antes: STORE em ADABAS-SOCIOS)
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
