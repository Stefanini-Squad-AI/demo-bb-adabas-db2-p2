DEFINE DATA
LOCAL USING STFBKSC00
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #DATA-ISO (A10)
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXTO MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* DBATDP-18 INICIO - Migracao ADABAS -> COBOL/DB2 (FIND)
STFBKSC00-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
STFBKSC00-ACAO := 'C'
CALLNAT 'STFSC00C' STFBKSC00-COMUNICACAO
IF STFBKSC00-RETORNO = 100
  PERFORM INCLUI-SOCIO
  REINPUT #TEXTO
END-IF
IF STFBKSC00-RETORNO = 0
  #TEXTO := 'Sócio já cadastrado.'
END-IF
* DBATDP-18 FIM - Migracao ADABAS -> COBOL/DB2 (FIND)
*
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
*   IF NO
*     PERFORM INCLUI-SOCIO
*     ESCAPE BOTTOM
*   END-NOREC
*   #TEXTO := 'Sócio já cadastrado.'
* END-FIND
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
STFBKSC00-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF STFBKSC00-NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF STFBKSC00-CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *STFBKSC00-CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *STFBKSC00-CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE EDITED *DATX (EM=YYYYMMDD) TO #DATA-ISO
  WITH MASK '9999-99-99'
STFBKSC00-DATA-CADASTRO := #DATA-ISO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' STFBKSC00-CATG-SOCIO STFBKSC00-VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
STFBKSC00-PAGAMENTO-OK(1) := 'T'
STFBKSC00-PAGAMENTO-OK(2:12) := 'F'
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE (EM=YYYYMMDD) TO #DATA-ISO
    WITH MASK '9999-99-99'
  STFBKSC00-DATA-VENCIMENTO(#INDICE) := #DATA-ISO
  STFBKSC00-SEQ-PERIODICO(#INDICE) := #INDICE
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
STFBKSC00-INDI-DIVIDA := 'F'
RESET STFBKSC00-DATA-BAIXA STFBKSC00-HORA-BAIXA
*
IF STFBKSC00-OBSV-SOCIO EQ ' '
  STFBKSC00-OBSV-SOCIO := 'Novo sócio'
END-IF
*
STFBKSC00-QTD-PERIODICO := 12
STFBKSC00-ACAO := 'I'
* DBATDP-18 INICIO - Migracao ADABAS -> COBOL/DB2 (STORE)
CALLNAT 'STFSC00I' STFBKSC00-COMUNICACAO
IF STFBKSC00-RETORNO = 803
  REINPUT FULL 'Erro: chave duplicada ao incluir sócio.'
END-IF
IF STFBKSC00-RETORNO NE 0
  COMPRESS 'ERRO DB2 ' STFBKSC00-RETORNO INTO #TEXTO
  REINPUT #TEXTO
END-IF
* DBATDP-18 FIM - Migracao ADABAS -> COBOL/DB2 (STORE)
*
* STORE
#TEXTO := 'Novo sócio incluído com sucesso.'
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
