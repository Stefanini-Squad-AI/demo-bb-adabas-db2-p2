DEFINE DATA
LOCAL USING BKSTFSOC
LOCAL
1 #MAPA
  2 #RG-CONSULTA (N7) 
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
*
1 NOME-SOCIO-PRINCIPAL (A40)
1 CATG-SOCIO (I2)
1 VALR-MENSALIDADE (N7,2/1:12)
1 DATA-VENCIMENTO (D/1:12)
1 PAGAMENTO-OK (L/1:12)
1 INDI-DIVIDA (L)
1 DATA-BAIXA (D)
1 HORA-BAIXA (A12)
1 OBSV-CLIENTE (A500)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* REPLACED ADABAS: FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA (DBATDP-8)
MOVE 'C' TO WS-OPER-CD
RESET WS-RETURN-CODE
MOVE #RG-CONSULTA TO WS-NUMB-SOCIO-PRINCIPAL
CALLNAT 'STFSC00C' BOOK-STF-SOCIOS
IF WS-RETURN-CODE = 1
  PERFORM INCLUI-SOCIO
ELSE
  IF WS-RETURN-CODE = 0
    #TEXTO := 'Sócio já cadastrado.'
  ELSE
    #TEXTO := 'Erro ao consultar sócio.'
  END-IF
END-IF

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
MOVE #RG-CONSULTA TO WS-NUMB-SOCIO-PRINCIPAL

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  DATA-VENCIMENTO(#INDICE) := #DATA-HOJE
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET INDI-DIVIDA DATA-BAIXA HORA-BAIXA
*
IF OBSV-CLIENTE EQ ' '
  OBSV-CLIENTE := 'Novo sócio'
END-IF
*
* REPLACED ADABAS: STORE (DBATDP-8) — persistência via STFSC00I / DB2
MOVE 'I' TO WS-OPER-CD
RESET WS-RETURN-CODE
MOVE #RG-CONSULTA TO WS-NUMB-SOCIO-PRINCIPAL
MOVE NOME-SOCIO-PRINCIPAL TO WS-NOME-SOCIO-PRINCIPAL
MOVE CATG-SOCIO TO WS-CATG-SOCIO
MOVE EDITED *DATX TO WS-DATA-CADASTRO-IN (EM=YYYYMMDD)
MOVE '        ' TO WS-DATA-BAIXA-IN
MOVE '        ' TO WS-HORA-BAIXA
MOVE OBSV-CLIENTE TO WS-OBSV-SOCIO
FOR #INDICE = 1 TO 12
  MOVE EDITED DATA-VENCIMENTO(#INDICE) TO WS-DATA-VENCIMENTO-IN(#INDICE) (EM=YYYYMMDD)
  MOVE VALR-MENSALIDADE(#INDICE) TO WS-VALR-MENSALIDADE(#INDICE)
  IF PAGAMENTO-OK(#INDICE)
    MOVE 'Y' TO WS-PAGAMENTO-OK(#INDICE)
  ELSE
    MOVE 'N' TO WS-PAGAMENTO-OK(#INDICE)
  END-IF
END-FOR
CALLNAT 'STFSC00I' BOOK-STF-SOCIOS
DECIDE ON FIRST VALUE WS-RETURN-CODE
  VALUE 0
    IGNORE
  VALUE 2
    REINPUT FULL 'Sócio já existe (duplicidade).'
  NONE
    REINPUT FULL 'Erro ao incluir sócio.'
END-DECIDE
#TEXTO := 'Novo sócio incluído com sucesso.'
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
