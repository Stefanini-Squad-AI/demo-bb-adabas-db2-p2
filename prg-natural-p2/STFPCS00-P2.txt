DEFINE DATA
LOCAL USING "STFSC00-LOCAL"
LOCAL
*-- Migracao DBATDP-17: acesso Adabas substituido por Cobol/DB2
*1 SOCIO VIEW OF ADABAS-SOCIOS
*  2  NUMB-SOCIO-PRINCIPAL
*  2  NOME-SOCIO-PRINCIPAL
*  2  DATA-CADASTRO
*  2  C*PERIODICO-PAGAMENTO
*  2  VALR-MENSALIDADE (12)
*  2  DATA-VENCIMENTO (12)
*  2  PAGAMENTO-OK (12)
*  2  CATG-SOCIO
*  2  INDI-DIVIDA
*  2  DATA-BAIXA
*  2  HORA-BAIXA
*  2  OBSV-CLIENTE
*  2  SUPER1
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
*
* Campos vinculados ao MAPSOCIO (sincronizados com STFSC00-AREA)
1 NOME-SOCIO-PRINCIPAL (A40)
1 CATG-SOCIO (N2)
1 OBSV-CLIENTE (A500)
1 VALR-MENSALIDADE (1:12) (P6.2)
1 DATA-VENCIMENTO (1:12) (A10)
1 PAGAMENTO-OK (1:12) (A1)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
PERFORM SYNC-MAPA-PARA-BOOK
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
*-- FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
RESET STFSC00-AREA
STFSC00-OPERACAO := 'C'
STFSC00-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALL 'STFSC00C' STFSC00-AREA
DECIDE ON FIRST VALUE OF STFSC00-SQLCODE
  VALUE 100
    PERFORM INCLUI-SOCIO
  VALUE 0
    #TEXTO := 'Sócio já cadastrado.'
  VALUE 803
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2 ' STFSC00-SQLCODE INTO #TEXTO
    REINPUT #TEXTO
END-DECIDE
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE SYNC-MAPA-PARA-BOOK
MOVE NOME-SOCIO-PRINCIPAL TO STFSC00-NOME-SOCIO-PRINCIPAL
MOVE CATG-SOCIO TO STFSC00-CATG-SOCIO
MOVE OBSV-CLIENTE TO STFSC00-OBSV-CLIENTE
MOVE VALR-MENSALIDADE(*) TO STFSC00-VALR-MENSALIDADE(*)
END-SUBROUTINE
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
PERFORM SYNC-MAPA-PARA-BOOK
*
STFSC00-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
COMPRESS #ANO '-' #MES '-' #DIA INTO STFSC00-DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' STFSC00-CATG-SOCIO STFSC00-VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
STFSC00-PAGAMENTO-OK(1) := 'Y'
STFSC00-PAGAMENTO-OK(2:12) := 'N'
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  COMPRESS #ANO '-' #MES '-' #DIA
    INTO STFSC00-DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES := 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
STFSC00-INDI-DIVIDA := 'N'
RESET STFSC00-DATA-BAIXA STFSC00-HORA-BAIXA
*
IF OBSV-CLIENTE EQ ' '
  OBSV-CLIENTE := 'Novo sócio'
  STFSC00-OBSV-CLIENTE := OBSV-CLIENTE
END-IF
*
STFSC00-C-PERIODICO-PAGAMENTO := 12
STFSC00-OPERACAO := 'I'
PERFORM SYNC-MAPA-PARA-BOOK
*-- STORE
CALL 'STFSC00I' STFSC00-AREA
DECIDE ON FIRST VALUE OF STFSC00-SQLCODE
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 803
    #TEXTO := 'Sócio já cadastrado.'
  NONE VALUE
    COMPRESS 'ERRO DB2 ' STFSC00-SQLCODE INTO #TEXTO
    REINPUT #TEXTO
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
