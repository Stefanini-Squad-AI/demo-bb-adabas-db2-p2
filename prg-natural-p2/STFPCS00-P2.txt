DEFINE DATA
LOCAL USING STFSC00L
*
* --- ADABAS (substituído por COBOL/DB2 via STFSC00L) ---
* 1 SOCIO VIEW OF ADABAS-SOCIOS
*   2  NUMB-SOCIO-PRINCIPAL
*   2  NOME-SOCIO-PRINCIPAL
*   2  DATA-CADASTRO
*   2  C*PERIODICO-PAGAMENTO
*   2  VALR-MENSALIDADE (12)
*   2  DATA-VENCIMENTO (12)
*   2  PAGAMENTO-OK (12)
*   2  CATG-SOCIO
*   2  INDI-DIVIDA
*   2  DATA-BAIXA
*   2  HORA-BAIXA
*   2  OBSV-CLIENTE
*   2  SUPER1
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
  2 NOME-SOCIO-PRINCIPAL (A40)
  2 CATG-SOCIO (N2)
  2 OBSV-CLIENTE (A500)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* --- ADABAS: FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA ---
STFSC00L-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALLNAT 'STFSC00C' STFSC00L
*
IF STFSC00L-RETURN-CODE = '+100'
  PERFORM INCLUI-SOCIO
ELSE
  IF STFSC00L-RETURN-CODE = '+000'
    #TEXTO := 'Sócio já cadastrado.'
  ELSE
    COMPRESS 'Erro na consulta (' STFSC00L-RETURN-CODE ')' INTO #TEXTO
  END-IF
END-IF
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
STFSC00L-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
STFSC00L-NOME-SOCIO-PRINCIPAL := NOME-SOCIO-PRINCIPAL
*
IF STFSC00L-NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
STFSC00L-CATG-SOCIO := CATG-SOCIO
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE EDITED *DATX (EM=YYYY-MM-DD) TO STFSC00L-DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO STFSC00L-VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
STFSC00L-PAGAMENTO-OK(1) := TRUE
STFSC00L-PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := *DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE (EM=YYYY-MM-DD)
    TO STFSC00L-DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET STFSC00L-INDI-DIVIDA STFSC00L-DATA-BAIXA STFSC00L-HORA-BAIXA
C*STFSC00L-PERIODICO-PAGAMENTO := 12
*
IF OBSV-CLIENTE EQ ' '
  STFSC00L-OBSV-SOCIO := 'Novo sócio'
ELSE
  STFSC00L-OBSV-SOCIO := OBSV-CLIENTE
END-IF
*
* --- ADABAS: STORE ---
CALLNAT 'STFSC00I' STFSC00L
*
DECIDE ON FIRST VALUE OF STFSC00L-RETURN-CODE
  VALUE '+000'
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE '+803'
    #TEXTO := 'Sócio já cadastrado (chave duplicada).'
  NONE VALUE
    COMPRESS 'Erro na inclusão (' STFSC00L-RETURN-CODE ')' INTO #TEXTO
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
