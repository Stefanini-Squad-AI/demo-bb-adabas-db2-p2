DEFINE DATA
LOCAL USING STFSC00-LCL
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* OLD-ADABAS: FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
MOVE #RG-CONSULTA TO NUMB-SOCIO-PRINCIPAL
CALL 'STFSC00C' SOCIOS-COMM
IF RETURN-CODE = 1
  PERFORM INCLUI-SOCIO
  ESCAPE BOTTOM
ELSE IF RETURN-CODE = 2
  #TEXTO := 'Erro ao acessar base de dados.'
  REINPUT #TEXTO
ELSE
  #TEXTO := 'Sócio já cadastrado.'
END-IF

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
MOVE #RG-CONSULTA TO NUMB-SOCIO-PRINCIPAL

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
MOVE DATX(EM=YYYYMMDD) TO #DATA-HOJE
PERFORM FORMAT-DATE-CADASTRO
MOVE #DATA-HOJE TO DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := 1
PAGAMENTO-OK(2:11) := 0
*
#DIA := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  PERFORM FORMAT-DATE-VENCIMENTO
  MOVE #DATA-HOJE TO DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET DATA-BAIXA HORA-BAIXA
*
IF OBSV-SOCIO EQ ' '
  OBSV-SOCIO := 'Novo sócio'
END-IF
*
* OLD-ADABAS: STORE
CALL 'STFSC00I' SOCIOS-COMM
IF RETURN-CODE EQ 0
  #TEXTO := 'Novo sócio incluído com sucesso.'
ELSE
  #TEXTO := 'Erro ao incluir novo sócio.'
END-IF
*
END-SUBROUTINE

* Formata DATA-CADASTRO de YYYYMMDD para YYYY-MM-DD
DEFINE FORMAT-DATE-CADASTRO
  COMPRESS #ANO '-' #MES '-' #DIA DELIMITED BY SIZE
    INTO DATA-CADASTRO
END-DEFINE

* Formata DATA-VENCIMENTO de YYYYMMDD para YYYY-MM-DD
DEFINE FORMAT-DATE-VENCIMENTO
  COMPRESS #ANO '-' #MES '-' #DIA DELIMITED BY SIZE
    INTO #DATA-HOJE
END-DEFINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
