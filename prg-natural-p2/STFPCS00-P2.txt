DEFINE DATA
LOCAL USING 'STFSCSOC'
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #DATA-DB2 (A10)
1 #HORA-DB2 (A5)
*
1 #MENSAGEM
  2 #TEXTO (A72)
*
* --- Legado ADABAS (DBATDP-17): view comentada; dados via STFSCSOC ---
* 1 SOCIO VIEW OF ADABAS-SOCIOS
*   2  NUMB-SOCIO-PRINCIPAL
*   2  NOME-SOCIO-PRINCIPAL
*   2  DATA-CADASTRO
*   2  C*PERIODICO-PAGAMENTO
*   2  VALR-MENSALIDADE (12)
*   2  DATA-VENCIMENTO (12)
*   2  PAGAMENTO-OK (12)
*   2  CATG-SOCIO
*   2  INDI-DIVIDA
*   2  DATA-BAIXA
*   2  HORA-BAIXA
*   2  OBSV-CLIENTE
*   2  SUPER1
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
RESET STFSCSOC-AREA
STFSCSOC-ACAO := 'C'
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALL 'STFSC00C' STFSCSOC-AREA
*
* --- Legado ADABAS (DBATDP-17) ---
* FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
*   IF NO
*     PERFORM INCLUI-SOCIO
*     ESCAPE BOTTOM
*   END-NOREC
*   #TEXTO := 'Sócio já cadastrado.'
* END-FIND
*
IF STFSCSOC-SQLCODE EQ 100
  PERFORM INCLUI-SOCIO
ELSE
  IF STFSCSOC-SQLCODE EQ 0
    #TEXTO := 'Sócio já cadastrado.'
  ELSE
    COMPRESS 'ERRO DB2' STFSCSOC-SQLCODE INTO #TEXTO
  END-IF
END-IF
*
REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
*
IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
MOVE EDITED #DATA-HOJE (EM=YYYY-MM-DD) TO DATA-CADASTRO
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := 'S'
PAGAMENTO-OK(2:11) := 'N'
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  MOVE EDITED #DATA-HOJE (EM=YYYY-MM-DD) TO DATA-VENCIMENTO(#INDICE)
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
INDI-DIVIDA := 'N'
RESET DATA-BAIXA HORA-BAIXA
*
IF OBSV-SOCIO EQ ' '
  OBSV-SOCIO := 'Novo sócio'
END-IF
*
C-PERIODICO-PAGAMENTO := 12
STFSCSOC-ACAO := 'I'
CALL 'STFSC00I' STFSCSOC-AREA
*
* --- Legado ADABAS (DBATDP-17) ---
* STORE
*
IF STFSCSOC-SQLCODE EQ 0
  #TEXTO := 'Novo sócio incluído com sucesso.'
ELSE
  IF STFSCSOC-SQLCODE EQ 803
    #TEXTO := 'Sócio já cadastrado.'
  ELSE
    COMPRESS 'ERRO DB2' STFSCSOC-SQLCODE INTO #TEXTO
  END-IF
END-IF
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
