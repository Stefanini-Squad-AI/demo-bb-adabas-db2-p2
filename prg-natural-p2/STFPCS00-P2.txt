DEFINE DATA
LOCAL USING 'STFSSOCI'
LOCAL
* ADABAS: 1 SOCIO VIEW OF ADABAS-SOCIOS (acesso direto removido - DBATDP-10)
1 SOCIO
  2  NUMB-SOCIO-PRINCIPAL
  2  NOME-SOCIO-PRINCIPAL
  2  DATA-CADASTRO
  2  C*PERIODICO-PAGAMENTO
  2  VALR-MENSALIDADE (12)
  2  DATA-VENCIMENTO (12)
  2  PAGAMENTO-OK (12)
  2  CATG-SOCIO
  2  INDI-DIVIDA
  2  DATA-BAIXA
  2  HORA-BAIXA
  2  OBSV-CLIENTE
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO
  2 #MES
  2 #DIA
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* ADABAS: FIND SOCIO WITH NUMB-SOCIO-PRINCIPAL EQ #RG-CONSULTA
RESET STFSSOCI
STFSSOCI-NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA
CALLNAT 'STFSC00C' STFSSOCI
DECIDE ON FIRST STFSSOCI-RETURN-CODE
  VALUE 0
    PERFORM MOVE-STFSSOCI-TO-SOCIO
    #TEXTO := 'Sócio já cadastrado.'
  VALUE 100
    PERFORM INCLUI-SOCIO
  NONE VALUE
    COMPRESS 'ERRO DB2/COBOL' STFSSOCI-RETURN-CODE INTO #TEXTO
    REINPUT #TEXTO
END-DECIDE

REINPUT #TEXTO
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
DATA-CADASTRO := *DATX
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*)
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  DATA-VENCIMENTO(#INDICE) := #DATA-HOJE
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET INDI-DIVIDA DATA-BAIXA HORA-BAIXA
*
IF OBSV-CLIENTE EQ ' '
  OBSV-CLIENTE := 'Novo sócio'
END-IF
*
* ADABAS: STORE
PERFORM MOVE-SOCIO-TO-STFSSOCI
CALLNAT 'STFSC00I' STFSSOCI
DECIDE ON FIRST STFSSOCI-RETURN-CODE
  VALUE 0
    #TEXTO := 'Novo sócio incluído com sucesso.'
  VALUE 803
    #TEXTO := 'Sócio já cadastrado (chave duplicada).'
  NONE VALUE
    COMPRESS 'ERRO DB2/COBOL' STFSSOCI-RETURN-CODE INTO #TEXTO
END-DECIDE
*
END-SUBROUTINE
* ----------------------------------------------------------------------
DEFINE MOVE-SOCIO-TO-STFSSOCI
STFSSOCI-NUMB-SOCIO-PRINCIPAL := NUMB-SOCIO-PRINCIPAL
STFSSOCI-NOME-SOCIO-PRINCIPAL := NOME-SOCIO-PRINCIPAL
MOVE EDITED DATA-CADASTRO (EM=YYYY-MM-DD) TO STFSSOCI-DATA-CADASTRO
C*STFSSOCI-PERIODICO-PAGAMENTO := 12
FOR #INDICE = 1 TO 12
  MOVE EDITED DATA-VENCIMENTO(#INDICE) (EM=YYYY-MM-DD)
    TO STFSSOCI-DATA-VENCIMENTO(#INDICE)
  STFSSOCI-VALR-MENSALIDADE(#INDICE) := VALR-MENSALIDADE(#INDICE)
  IF PAGAMENTO-OK(#INDICE)
    STFSSOCI-PAGAMENTO-OK(#INDICE) := '1'
  ELSE
    STFSSOCI-PAGAMENTO-OK(#INDICE) := '0'
  END-IF
END-FOR
STFSSOCI-CATG-SOCIO := CATG-SOCIO
IF INDI-DIVIDA
  STFSSOCI-INDI-DIVIDA := '1'
ELSE
  STFSSOCI-INDI-DIVIDA := '0'
END-IF
IF DATA-BAIXA NE 0
  MOVE EDITED DATA-BAIXA (EM=YYYY-MM-DD) TO STFSSOCI-DATA-BAIXA
ELSE
  STFSSOCI-DATA-BAIXA := ' '
END-IF
IF HORA-BAIXA NE 0
  MOVE EDITED HORA-BAIXA (EM=HH:MM) TO STFSSOCI-HORA-BAIXA
ELSE
  STFSSOCI-HORA-BAIXA := ' '
END-IF
STFSSOCI-OBSV-SOCIO := OBSV-CLIENTE
END-SUBROUTINE
* ----------------------------------------------------------------------
DEFINE MOVE-STFSSOCI-TO-SOCIO
NUMB-SOCIO-PRINCIPAL := STFSSOCI-NUMB-SOCIO-PRINCIPAL
NOME-SOCIO-PRINCIPAL := STFSSOCI-NOME-SOCIO-PRINCIPAL
MOVE EDITED STFSSOCI-DATA-CADASTRO TO DATA-CADASTRO (EM=YYYY-MM-DD)
C*PERIODICO-PAGAMENTO := C*STFSSOCI-PERIODICO-PAGAMENTO
FOR #INDICE = 1 TO C*STFSSOCI-PERIODICO-PAGAMENTO
  MOVE EDITED STFSSOCI-DATA-VENCIMENTO(#INDICE)
    TO DATA-VENCIMENTO(#INDICE) (EM=YYYY-MM-DD)
  VALR-MENSALIDADE(#INDICE) := STFSSOCI-VALR-MENSALIDADE(#INDICE)
  IF STFSSOCI-PAGAMENTO-OK(#INDICE) = '1'
    PAGAMENTO-OK(#INDICE) := TRUE
  ELSE
    PAGAMENTO-OK(#INDICE) := FALSE
  END-IF
END-FOR
CATG-SOCIO := STFSSOCI-CATG-SOCIO
IF STFSSOCI-INDI-DIVIDA = '1'
  INDI-DIVIDA := TRUE
ELSE
  INDI-DIVIDA := FALSE
END-IF
IF STFSSOCI-DATA-BAIXA NE ' '
  MOVE EDITED STFSSOCI-DATA-BAIXA TO DATA-BAIXA (EM=YYYY-MM-DD)
ELSE
  RESET DATA-BAIXA
END-IF
IF STFSSOCI-HORA-BAIXA NE ' '
  MOVE EDITED STFSSOCI-HORA-BAIXA TO HORA-BAIXA (EM=HH:MM)
ELSE
  RESET HORA-BAIXA
END-IF
OBSV-CLIENTE := STFSSOCI-OBSV-SOCIO
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
