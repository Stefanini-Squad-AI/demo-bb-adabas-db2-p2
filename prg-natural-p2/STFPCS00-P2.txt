DEFINE DATA
LOCAL USING "STFSCK00"
*
* [ADABAS-MIGRATED] Original ADABAS structure replaced by COBOL data access
*
1 #MAPA
  2 #RG-CONSULTA (N7)
  2 #DIA-VENCIAMENTO (N2)
*
1 #INDICE (N2)
1 #DATA-HOJE (A8)
1 REDEFINES #DATA-HOJE
  2 #ANO (A4)
  2 #MES (A2)
  2 #DIA (A2)
*
1 #MENSAGEM
  2 #TEXTO (A72)
END-DEFINE
*
SET KEY ALL
SET CONTROL 'M-2'
*
INPUT WITH TEXT #TEXT MARK #LOCAL MAP 'MAPSOCIO'
*
IF *PF-KEY EQ 'F3'
  INPUT NO ERASE 'Fim do processo. Aperte qualquer tecla para sair.'
  ESCAPE PROGRAM
END-IF
IF *PF-KEY NE 'ENTR'
  REINPUT FULL 'Consulte RG ou aperte F3 para sair.'
END-IF
IF #RG-CONSULTA EQ 0
  REINPUT FULL 'Informe RG do novo sócio' MARK *#RG-CONSULTA
END-IF
*
* [ADABAS] Original FIND replaced with CALLNAT 'STFSC00C'
MOVE SPACES TO SOCIOS-BOOK.
MOVE SPACES TO PAGAMENTO-BOOK.
MOVE 'C' TO OPERATION-CODE.
MOVE #RG-CONSULTA TO NUMB-SOCIO-PRINCIPAL.
CALLNAT 'STFSC00C' USING SOCIOS-BOOK PAGAMENTO-BOOK COMM-BOOK.

IF RETURN-CODE = 0
  #TEXTO := 'Sócio já cadastrado.'
  REINPUT #TEXTO
ELSE
  IF RETURN-CODE = 1
    PERFORM INCLUI-SOCIO
    ESCAPE BOTTOM
  ELSE
    #TEXTO := ERROR-MESSAGE
    REINPUT #TEXTO
  END-IF
END-IF
* ----------------------------------------------------------------------
DEFINE INCLUI-SOCIO
*
NUMB-SOCIO-PRINCIPAL := #RG-CONSULTA

IF NOME-SOCIO-PRINCIPAL EQ ' '
  REINPUT FULL 'Informe nome do novo Sócio.'
END-IF
*
DECIDE FOR FIRST VALUE OF *CATG-SOCIO
  VALUE 0 REINPUT FULL 'Informe a categoria.' MARK *CATG-SOCIO
  VALUE 1, 2  IGNORE
  NONE VALUE 'Categoria inválida.' MARK *CATG-SOCIO
END-DECIDE
*
IF #DIA-VENCIAMENTO EQ 0
 REINPUT FULL 'Informe a o dia do vencimento.'
END-IF
IF NOT #DIA-VENCIAMENTO EQ 1 OR EQ 5 OR EQ 15 OR EQ 20 OR EQ 25
 REINPUT FULL 'Dia do vencimento inválido.'
END-IF
*
DATA-CADASTRO := *DATX
*
* Busca valor da mensalidade por categoria do novo cliente
CALLNAT 'VERVALOR' CATG-SOCIO VALR-MENSALIDADE(*) 
*
* Todo o novo sócio já precisa ter a primeira mensalidade paga
PAGAMENTO-OK(1) := TRUE
PAGAMENTO-OK(2:11) := FALSE
*
#DATA-HOJE := DATX (EM=YYYYMMDD)
#DIA       := #DIA-VENCIAMENTO
*
FOR #INDICE = 1 TO 12
  DATA-VENCIMENTO(#INDICE) := #DATA-HOJE
  ADD 1 TO #MES
  IF #MES > 12
    #MES = 1
    ADD 1 TO #ANO
  END-IF
END-FOR
*
RESET INDI-DIVIDA DATA-BAIXA HORA-BAIXA
*
IF OBSV-SOCIO EQ ' '
  OBSV-SOCIO := 'Novo sócio'
END-IF
*
* [ADABAS] Original STORE replaced with CALLNAT 'STFSC00I'
MOVE 'I' TO OPERATION-CODE.
CALLNAT 'STFSC00I' USING SOCIOS-BOOK PAGAMENTO-BOOK COMM-BOOK.

IF RETURN-CODE = 0
  #TEXTO := 'Novo sócio incluído com sucesso.'
ELSE
  #TEXTO := ERROR-MESSAGE
END-IF
*
END-SUBROUTINE
* ----------------------------------------------------------------------
ON ERROR
  COMPRESS 'ERRO ' *ERROR ' NA LINHA ' *ERROR-LINE INTO #TEXTO
  REINPUT #TEXTO
END-ERROR
END
